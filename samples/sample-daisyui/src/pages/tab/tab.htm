Dynamic Tab Management

Add, update, and remove tabs programmatically

const tabComponent = document.querySelector('baz-tab');
// Add a new tab with icon
tabComponent.addTab(
  'settings',
  'Settings',
  '<div class="p-4"><h3>Settings</h3></div>',
  'settings',
  true  // activate immediately
);
// Update existing tab
tabComponent.updateTab(
  'settings',
  'App Settings',  // new title
  '<div>Updated content</div>',
  'gear'  // new icon
);
// Remove a tab
tabComponent.removeTab('settings');
// Check if tab exists
if (tabComponent.hasTab('settings')) {
  console.log('Tab exists');
}

Event Handling

Listen to tab changes and property updates

const tabs = document.querySelector('#myTabs');
// Listen to property changes
tabs.addEventListener('property-changed', (e) => {
  if (e.detail.propertyName === 'ActiveTabId') {
    console.log('Active tab changed to:', e.detail.value);
    
    // Analytics tracking
    trackTabView(e.detail.value);
    
    // Load content dynamically
    loadTabContent(e.detail.value);
  }
});
// Get current state
console.log('Total tabs:', tabs.getTabCount());
console.log('Active tab:', tabs.ActiveTabId);
// Programmatic tab activation
tabs.ActiveTabId = 'profile';

Best Practices

Tips

Use Unique Tab IDs

Always use unique, descriptive tab IDs for better debugging and maintenance

Icons Enhance UX

Add icons to tabs for better visual identification and user experience

Lazy Load Content

For heavy content, load tab content only when the tab becomes active

Limit Tab Count

Keep the number of visible tabs reasonable (5-8 max) for better usability
