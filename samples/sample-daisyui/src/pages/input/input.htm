Size Variants

      
Button size is automatically one level below input size (xl→lg, lg→md, md→sm, sm→xs, xs→xs)

Programmatic Button API

      
      

        
// Get the input component

        
const input = document.querySelector('baz-input');

        

        
// Set buttons programmatically

        
input.setButtons([

        
  {

        
    name: 'search',

        
    icon: 'search',

        
    description: 'Search',

        
    color: 'ghost',

        
    onClick: (e) => console.log('Search clicked')

        
  },

        
  {

        
    name: 'submit',

        
    icon: 'arrow-right',

        
    description: 'Submit',

        
    color: 'primary',

        
    onClick: (e) => console.log('Submit clicked')

        
  }

        
]);

      

    

  


  
  

    

      
Attributes

      
      

        

          
            

              Attribute
              Type
              Default
              Description
            

          
          
            

              value
              string
              ""
              Input value (two-way bindable)
            

            

              label
              string
              ""
              Legend text (hidden if empty)
            

            

              description
              string
              ""
              Description text below label (hidden if empty)
            

            

              placeholder
              string
              "Type here..."
              Placeholder text
            

            

              prefix
              string
              ""
              Text before input (hidden if empty)
            

            

              suffix
              string
              ""
              Text after input (hidden if empty)
            

            

              help-text
              string
              ""
              Help text below input (hidden if empty)
            

            

              color
              string
              "neutral"
              primary | secondary | accent | info | success | warning | error
            

            

              size
              string
              "md"
              xs | sm | md | lg (buttons are one size smaller)
            

            

              type
              string
              "text"
              HTML input type
            

            

              disabled
              boolean
              false
              Disables the input
            

            

              readonly
              boolean
              false
              Makes the input readonly
            

          
        

      

      
      
Button Interface (IBazInputButton)

      

        

          
            

              Property
              Type
              Required
              Description
            

          
          
            

              name
              string
              Yes
              Unique button identifier
            

            

              icon
              string
              Yes
              Icon name for baz-icon
            

            

              description
              string
              No
              Tooltip text
            

            

              color
              string
              No
              ghost | primary | secondary | accent | etc.
            

            

              onClick
              function
              No
              Click handler (event: MouseEvent) => void
