@CustomElement

This decorator registers a class as a custom HTML element. It must be applied to the class declaration and defines the tag name used in HTML. 


@CustomElement(tagName: string)
  • tagName
    - The custom element tag name. Must contain a hyphen (e.g., "my-component").

Important: Custom element names must contain at least one hyphen (-) to distinguish them from standard HTML elements. 


import { BazlamaWebComponent, CustomElement } from "bazlama-web-component"

@CustomElement("my-greeting")
class MyGreeting extends BazlamaWebComponent {
    constructor() {
        super()
        this.InitBazlamaWebComponent()
    }
    
    getRenderTemplate() {
        return `<h1>Hello, World!</h1>`
    }
}

// Usage in HTML:
// <my-greeting></my-greeting>
        

import { BazlamaWebComponent, CustomElement, ShadowRootMode } from "bazlama-web-component"

// With Shadow DOM (default - closed)
@CustomElement("shadow-component")
class ShadowComponent extends BazlamaWebComponent {
    constructor() {
        super() // Uses closed shadow DOM by default
        this.InitBazlamaWebComponent()
    }
}

// Without Shadow DOM
@CustomElement("light-component")
class LightComponent extends BazlamaWebComponent {
    constructor() {
        super(ShadowRootMode.None) // Renders directly to element
        this.InitBazlamaWebComponent()
    }
}

// Open Shadow DOM (accessible via element.shadowRoot)
@CustomElement("open-shadow-component")
class OpenShadowComponent extends BazlamaWebComponent {
    constructor() {
        super(ShadowRootMode.Open)
        this.InitBazlamaWebComponent()
    }
}
