@ChangeHooks

This decorator registers hook functions that are called whenever a property changes. Hooks are the primary way to update the DOM in response to property changes. 


@ChangeHooks(hooks: TPropertyChangeHook[])

type TPropertyChangeHook = (
    bazComponent: BazlamaWebComponent,
    value: TPropertyValueType,
    propertyDefine: PropertyDefine,
    oldValue: TPropertyValueType
) => void
  • hooks
    - Array of hook functions to execute on property change.
  • useElementText(query, prefix, suffix) - Updates element's text content
  • useElementTextWithFunction(query, fn) - Updates text with custom formatting
  • useElementInputValue(query) - Syncs with input element value
  • useElementAttribute(query, attr) - Updates element attribute
  • useElementProperty(query, prop) - Updates element property
  • useElementStyle(query, style, suffix) - Updates inline style
  • useAddRemoveClass(options) - Adds/removes CSS classes
  • useToggleClass(query, className) - Toggles class based on boolean
  • useSwitchClass(query, prefix) - Switches between classes
  • useCustomHook(query, fn) - Custom logic per element
  • useFunction(fn) - General custom logic

@CustomElement("product-card")
class ProductCard extends BazlamaWebComponent {
    // Multiple hooks on one property
    @ChangeHooks([
        useElementText(".price-display", "$"),
        useElementAttribute(".product", "data-price"),
        useFunction((comp, value) => {
            console.log(`Price changed to: ${value}`)
        })
    ])
    @Attribute("price", true)
    @Property({ defaultValue: 0 })
    price: number = 0
    
    // Conditional class based on boolean
    @ChangeHooks([useToggleClass(".stock-badge", "out-of-stock")])
    @Attribute("in-stock", true)
    @Property({ defaultValue: true })
    inStock: boolean = true
}
